\NewDocumentCommand{\chemPolygone}{O{} r()}{
  \pgfkeys{/chemPoly, defaut, #1}
  \node [
    regular polygon, fill = \chemCouleur,
    regular polygon sides = \chemBords, 
    inner sep = \fpeval{\chemAtomSep * \chemBords * 0.103} pt,
    rotate = \chemRotation,
  ] at (#2) {};
}
